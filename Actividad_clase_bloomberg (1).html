# Step 1: Simulate a Random Walk
def random_walk_simulation(initial_price, steps, scale=1):
    random_steps = np.random.normal(loc=0, scale=scale, size=steps)
    random_steps[0] = 0  # Start at initial price
    return initial_price + np.cumsum(random_steps)

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, coint
from statsmodels.tsa.arima.model import ARIMA
import warnings
warnings.filterwarnings("ignore")

# Set random seed for reproducibility
np.random.seed(42)

# Datos originales
data_3min = {
    'Accin 1': {'Fecha': ['2025-03-07 11:27:00', '2025-03-07 11:24:00', '2025-03-05 08:30:00'], 'Precio': [193.913, 194.0, 205.12]},
    'Accin 2': {'Fecha': ['2025-03-07 11:36:00', '2025-03-07 11:33:00', '2025-03-04 08:30:00'], 'Precio': [109.4302, 109.5, 111.895]},
    'Accin 3': {'Fecha': ['2025-03-07 11:36:00', '2025-03-07 11:33:00', '2025-03-04 08:30:00'], 'Precio': [388.8, 388.75, 381.71]}
}

data_30min = {
    'Accin 1': {'Fecha': ['2025-03-07 11:30:00', '2025-03-07 11:00:00', '2025-02-20 08:30:00'], 'Precio': [194.08, 194.5, 222.83]},
    'Accin 2': {'Fecha': ['2025-03-07 11:30:00', '2025-03-07 11:00:00', '2025-02-24 08:30:00'], 'Precio': [109.5, 110.0, 214.53]},
    'Accin 3': {'Fecha': ['2025-03-07 11:30:00', '2025-03-07 11:00:00', '2025-02-20 08:30:00'], 'Precio': [388.75, 389.0, 413.93]}
}

# Simular ms datos (50 puntos) basados en los originales
def simulate_data(original_data, freq, steps=50):
    simulated = {}
    for key in original_data:
        start_price = original_data[key]['Precio'][-1]
        dates = pd.date_range(start=original_data[key]['Fecha'][-1], periods=steps, freq=freq)
        prices = start_price + np.cumsum(np.random.normal(0, 1, steps))  # Random walk con ruido
        simulated[key] = pd.DataFrame({'Fecha': dates, 'Precio': prices}).set_index('Fecha')
    return simulated

prices_3min = simulate_data(data_3min, '3min')
prices_30min = simulate_data(data_30min, '30min')

# Step 1: Unit Root Test (ADF)
def unit_root_test(series, name):
    result = adfuller(series.dropna())
    print(f"\nUnit Root Test (ADF) for {name}:")
    print(f"ADF Statistic: {result[0]:.4f}")
    print(f"p-value: {result[1]:.4f}")
    print("Critical Values:", {k: round(v, 4) for k, v in result[4].items()})
    if result[1] > 0.05:
        print(f"{name} likely has a unit root (non-stationary).")
    else:
        print(f"{name} does not have a unit root (stationary).")

# Calcular rendimientos
returns_3min = {key: prices_3min[key]['Precio'].pct_change().dropna() for key in prices_3min}
returns_30min = {key: prices_30min[key]['Precio'].pct_change().dropna() for key in prices_30min}

# Pruebas ADF
for key in prices_3min:
    unit_root_test(prices_3min[key]['Precio'], f"{key} Prices (3 min)")
    unit_root_test(returns_3min[key], f"{key} Returns (3 min)")
for key in prices_30min:
    unit_root_test(prices_30min[key]['Precio'], f"{key} Prices (30 min)")
    unit_root_test(returns_30min[key], f"{key} Returns (30 min)")

# Step 2: Fit ARMA Model (ARMA(1,1))
def fit_arma(series, name):
    try:
        model = ARIMA(series, order=(1, 0, 1))
        fit = model.fit()
        print(f"\nARMA(1,1) Model Summary for {name}:")
        print(fit.summary().tables[1])
        return fit
    except Exception as e:
        print(f"Error fitting ARMA model for {name}: {e}")
        return None

arma_3min = {key: fit_arma(returns_3min[key], f"{key} Returns (3 min)") for key in returns_3min}
arma_30min = {key: fit_arma(returns_30min[key], f"{key} Returns (30 min)") for key in returns_30min}

# Step 3: Cointegration Test
def cointegration_test(series1, series2, name1, name2):
    score, p_value, _ = coint(series1, series2)
    print(f"\nCointegration Test between {name1} and {name2}:")
    print(f"Cointegration Statistic: {score:.4f}")
    print(f"p-value: {p_value:.4f}")
    if p_value < 0.05:
        print(f"{name1} and {name2} are likely cointegrated.")
    else:
        print(f"No evidence of cointegration between {name1} and {name2}.")

# Pruebas de cointegracin entre pares de acciones (solo precios)
for i, key1 in enumerate(prices_3min):
    for key2 in list(prices_3min.keys())[i+1:]:
        cointegration_test(prices_3min[key1]['Precio'], prices_3min[key2]['Precio'],
                          f"{key1} (3 min)", f"{key2} (3 min)")
for i, key1 in enumerate(prices_30min):
    for key2 in list(prices_30min.keys())[i+1:]:
        cointegration_test(prices_30min[key1]['Precio'], prices_30min[key2]['Precio'],
                          f"{key1} (30 min)", f"{key2} (30 min)")
