import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller, coint
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.tsa.ar_model import AutoReg
import yfinance as yf
import warnings
warnings.filterwarnings("ignore")

# Set random seed for reproducibility
np.random.seed(42)

# Step 1: Fetch stock data for V and MA
start_date = "2019-11-29"
end_date = "2025-03-14"  # Current date

print("Fetching data for V...")
v_data = yf.download("V", start=start_date, end=end_date)
print("Fetching data for MA...")
ma_data = yf.download("MA", start=start_date, end=end_date)

# Extract the 'Close' column
v_prices = v_data["Close"].dropna()
ma_prices = ma_data["Close"].dropna()
print("V Prices Head:\n", v_prices.head())
print("MA Prices Head:\n", ma_prices.head())

# Calculate returns
v_returns = v_prices.pct_change().dropna()
ma_returns = ma_prices.pct_change().dropna()

# Step 2: Unit Root Test (ADF)
def unit_root_test(series, name):
    result = adfuller(series.dropna())
    print(f"\nUnit Root Test (ADF) for {name}:")
    print(f"ADF Statistic: {result[0]:.4f}")
    print(f"p-value: {result[1]:.4f}")
    print("Critical Values:", {k: round(v, 4) for k, v in result[4].items()})
    if result[1] > 0.05:
        print(f"{name} likely has a unit root (non-stationary).")
    else:
        print(f"{name} does not have a unit root (stationary).")

unit_root_test(v_prices, "V Prices")
unit_root_test(ma_prices, "MA Prices")
unit_root_test(v_returns, "V Returns")
unit_root_test(ma_returns, "MA Returns")

# Step 3: Fit AR Model
def fit_ar(series, name, lags=1):
    try:
        model = AutoReg(series, lags=lags)
        fit = model.fit()
        print(f"\nAR({lags}) Model Summary for {name}:")
        print(fit.summary().tables[1])
        return fit
    except Exception as e:
        print(f"Error fitting AR model for {name}: {e}")
        return None

ar_v = fit_ar(v_returns, "V Returns", lags=1)
ar_ma = fit_ar(ma_returns, "MA Returns", lags=1)

# Step 4: Fit ARMA Model
def fit_arma(series, name, order=(1, 1)):
    try:
        model = ARIMA(series, order=(order[0], 0, order[1]))  # (p, d, q) with d=0 for ARMA
        fit = model.fit()
        print(f"\nARMA({order[0]},{order[1]}) Model Summary for {name}:")
        print(fit.summary().tables[1])
        return fit
    except Exception as e:
        print(f"Error fitting ARMA model for {name}: {e}")
        return None

arma_v = fit_arma(v_returns, "V Returns", order=(1, 1))
arma_ma = fit_arma(ma_returns, "MA Returns", order=(1, 1))

# Step 5: Fit ARIMA Model
def fit_arima(series, name, order=(1, 1, 1)):
    try:
        model = ARIMA(series, order=order)  # (p, d, q)
        fit = model.fit()
        print(f"\nARIMA({order[0]},{order[1]},{order[2]}) Model Summary for {name}:")
        print(fit.summary().tables[1])
        return fit
    except Exception as e:
        print(f"Error fitting ARIMA model for {name}: {e}")
        return None

arima_v = fit_arima(v_returns, "V Returns", order=(1, 0, 1))  # d=0 asumiendo estacionariedad
arima_ma = fit_arima(ma_returns, "MA Returns", order=(1, 0, 1))

# Step 6: Cointegration Test
score, p_value, _ = coint(v_prices, ma_prices)
print(f"\nCointegration Test between V and MA:")
print(f"Cointegration Statistic: {score:.4f}")
print(f"p-value: {p_value:.4f}")
if p_value < 0.05:
    print("V and MA are likely cointegrated.")
else:
    print("No evidence of cointegration between V and MA.")

# Step 7: Forecasting
forecast_steps = 30  # 30 days forecast

# AR Forecast
if ar_v is not None:
    ar_v_forecast = ar_v.forecast(steps=forecast_steps)
    ar_ma_forecast = ar_ma.forecast(steps=forecast_steps)
else:
    ar_v_forecast = pd.Series()
    ar_ma_forecast = pd.Series()

# ARMA Forecast
if arma_v is not None:
    arma_v_forecast = arma_v.forecast(steps=forecast_steps)
    arma_ma_forecast = arma_ma.forecast(steps=forecast_steps)
else:
    arma_v_forecast = pd.Series()
    arma_ma_forecast = pd.Series()

# ARIMA Forecast
if arima_v is not None:
    arima_v_forecast = arima_v.forecast(steps=forecast_steps)
    arima_ma_forecast = arima_ma.forecast(steps=forecast_steps)
else:
    arima_v_forecast = pd.Series()
    arima_ma_forecast = pd.Series()

# Create forecast index
forecast_index = pd.date_range(start=v_returns.index[-1], periods=forecast_steps + 1, freq='B')[1:]

# Step 8: Plotting
plt.figure(figsize=(15, 12))

# Prices
plt.subplot(4, 1, 1)
plt.plot(v_prices.index, v_prices, label="V Prices", color="blue")
plt.plot(ma_prices.index, ma_prices, label="MA Prices", color="orange")
plt.title("Stock Prices (V and MA)")
plt.xlabel("Date")
plt.ylabel("Price")
plt.legend()

# Returns
plt.subplot(4, 1, 2)
plt.plot(v_returns.index, v_returns, label="V Returns", color="blue")
plt.plot(ma_returns.index, ma_returns, label="MA Returns", color="orange")
plt.title("Returns (V and MA)")
plt.xlabel("Date")
plt.ylabel("Returns")
plt.legend()

# AR Forecast
plt.subplot(4, 1, 3)
plt.plot(v_returns.index[-50:], v_returns[-50:], label="V Actual Returns", color="blue")
plt.plot(ma_returns.index[-50:], ma_returns[-50:], label="MA Actual Returns", color="orange")
plt.plot(forecast_index, ar_v_forecast, label="V AR Forecast", color="blue", linestyle="--")
plt.plot(forecast_index, ar_ma_forecast, label="MA AR Forecast", color="orange", linestyle="--")
plt.title("AR Forecast (Last 50 Days + 30 Days Prediction)")
plt.xlabel("Date")
plt.ylabel("Returns")
plt.legend()

# ARIMA Forecast
plt.subplot(4, 1, 4)
plt.plot(v_returns.index[-50:], v_returns[-50:], label="V Actual Returns", color="blue")
plt.plot(ma_returns.index[-50:], ma_returns[-50:], label="MA Actual Returns", color="orange")
plt.plot(forecast_index, arima_v_forecast, label="V ARIMA Forecast", color="blue", linestyle="--")
plt.plot(forecast_index, arima_ma_forecast, label="MA ARIMA Forecast", color="orange", linestyle="--")
plt.title("ARIMA Forecast (Last 50 Days + 30 Days Prediction)")
plt.xlabel("Date")
plt.ylabel("Returns")
plt.legend()

plt.tight_layout()
plt.show()
