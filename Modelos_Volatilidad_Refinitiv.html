# Step 2: Unit Root Test (Augmented Dickey-Fuller Test)
def adf_test(series, title=''):
    print(f'ADF Test on {title}')
    result = adfuller(series.dropna(), autolag='AIC')
    print(f'ADF Statistic: {result[0]}')
    print(f'p-value: {result[1]}')
    print(f'Critical Values: {result[4]}')
    if result[1] <= 0.05:
        print("Reject the null hypothesis - Series is stationary")
    else:
        print("Fail to reject the null hypothesis - Series is non-stationary")
    print('\n')

# Step 3: Stationarity Check and Differencing
def check_stationarity(series, title=''):
    adf_test(series, title)
    if adfuller(series.dropna())[1] > 0.05:
        print(f'Differencing {title}...')
        diff_series = series.diff().dropna()
        adf_test(diff_series, f'{title} (Differenced)')
        return diff_series
    return series

# Step 4: ARIMA Model and Forecasting
def fit_arima_and_forecast(series, title='', forecast_steps=10):
    print(f'Fitting ARIMA model for {title}')
    # Use auto_arima to automatically select the best ARIMA parameters
    model = auto_arima(series.dropna(), seasonal=False, stepwise=True, trace=True)
    print(model.summary())

    # Fit the model
    fitted_model = model.fit(series.dropna())

    # Forecast
    forecast = fitted_model.predict(n_periods=forecast_steps)

    # Plotting
    plt.figure(figsize=(10, 6))
    plt.plot(series, label='Original')
    plt.plot(pd.Series(forecast, index=range(len(series), len(series) + forecast_steps)), label='Forecast')
    plt.title(f'{title} Forecast')
    plt.legend()
    plt.show()

    return forecast
