def adf_test(series, title=''):
    result = adfuller(series.dropna())
    print(f'Test ADF en {title}')
    print(f'Estadístico ADF: {result[0]}')
    print(f'p-valor: {result[1]}')
    print(f'Valores Críticos: {result[4]}')
    if result[1] <= 0.05:
        print("Rechazamos la hipótesis nula - La serie es estacionaria")
    else:
        print("No rechazamos la hipótesis nula - La serie no es estacionaria, considere diferenciar")
